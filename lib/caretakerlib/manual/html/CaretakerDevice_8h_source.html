1 /************************************************************************************************
2 * @file CaretakerDevice.h
3 * Header file defining a managed common language runtime (CLR) class to allow .NET applications
4 * to monitor the Caretaker device. It imports the native (unmanaged libcaretaker_dynamic.dll)

7 * For portability with the various .NET languages (C#, Visual Basic, Visual C++, etc), only
8 * the .NET Framework types are to be exposed to the application. You can find the .NET Framework

1600  ref class WaveformRingBuffer {
1601  public:
1602  WaveformRingBuffer(int capacity) {
1603  mFront = mBack = mSize = 0;
1604  mCapacity = capacity;
1605  mRing = gcnew array<WaveformDataPoints^>(capacity);
1606  mMutex = gcnew System::Threading::Mutex();
1607  }
1608 
1609  WaveformDataPoints^ PopFront() {
1610  mMutex->WaitOne();
1611  WaveformDataPoints^ e = nullptr;
1612  if (!Empty()) {
1613  e = mRing[mFront];
1614  mSize -= 1;
1615  if (Empty()) {
1616  mFront = mBack = 0;
1617  }
1618  else {
1619  mFront = (mFront + 1) % mCapacity;
1620  }
1621  }
1622  mMutex->ReleaseMutex();
1623  return e;
1624  }
1625 
1626  void PushBack(WaveformDataPoints^ e) {
1627  mMutex->WaitOne();
1628  if (Empty()) {
1629  mBack = mFront = 0;
1630  }
1631  else {
1632  mBack = (mBack + 1) % mCapacity;
1633  }
1634 
1635  if (Full()) {
1636  mFront = (mFront + 1) % mCapacity;
1637  }
1638  else {
1639  mSize += 1;
1640  }
1641 
1642  mRing[mBack] = e;
1643  mMutex->ReleaseMutex();
1644  }
1645 
1646  void Clear() {
1647  mFront = mBack = mSize = 0;
1648  }
1649 
1650  bool Empty() {
1651  return (mSize == 0);
1652  }
1653 
1654  bool Full() {
1655  return (mSize == mCapacity);
1656  }
1657 
1658  int Size() {
1659  return mSize;
1660  }
1661 
1662  int Capacity() {
1663  return mCapacity;
1664  }
1665 
1666  int mFront; // front entry
1667  int mBack; // back entry
1668  int mSize; // count of entries in the ring
1669  int mCapacity; // ring capacity
1670  array<WaveformDataPoints^>^ mRing;
1671  System::Threading::Mutex^ mMutex;
1672  };
