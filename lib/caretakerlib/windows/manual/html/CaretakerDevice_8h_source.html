1 /************************************************************************************************
2 * @file CaretakerDevice.h
3 * Header file defining a managed common language runtime (CLR) class to allow .NET applications
4 * to monitor the Caretaker device. It imports the native (unmanaged libcaretaker_dynamic.dll)
5 * and uses Windows platform invoke P/Invoke to call into the library.
6 *
7 * For portability with the various .NET languages (C#, Visual Basic, Visual C++, etc), only
8 * the .NET Framework types are to be exposed to the application. You can find the .NET Framework
9 * Equivalents to C++ Native Types (C++/CLI) here on the Microsoft website
10 * https://msdn.microsoft.com/en-us/library/0wf2yk2k.aspx.

1378  ref class WaveformRingBuffer {
1379  public:
1380  WaveformRingBuffer(int capacity) {
1381  mFront = mBack = mSize = 0;
1382  mCapacity = capacity;
1383  mRing = gcnew array<WaveformDataPoints^>(capacity);
1384  mMutex = gcnew System::Threading::Mutex();
1385  }
1386 
1387  WaveformDataPoints^ PopFront() {
1388  mMutex->WaitOne();
1389  WaveformDataPoints^ e = nullptr;
1390  if (!Empty()) {
1391  e = mRing[mFront];
1392  mSize -= 1;
1393  if (Empty()) {
1394  mFront = mBack = 0;
1395  }
1396  else {
1397  mFront = (mFront + 1) % mCapacity;
1398  }
1399  }
1400  mMutex->ReleaseMutex();
1401  return e;
1402  }
1403 
1404  void PushBack(WaveformDataPoints^ e) {
1405  mMutex->WaitOne();
1406  if (Empty()) {
1407  mBack = mFront = 0;
1408  }
1409  else {
1410  mBack = (mBack + 1) % mCapacity;
1411  }
1412 
1413  if (Full()) {
1414  mFront = (mFront + 1) % mCapacity;
1415  }
1416  else {
1417  mSize += 1;
1418  }
1419 
1420  mRing[mBack] = e;
1421  mMutex->ReleaseMutex();
1422  }
1423 
1424  void Clear() {
1425  mFront = mBack = mSize = 0;
1426  }
1427 
1428  bool Empty() {
1429  return (mSize == 0);
1430  }
1431 
1432  bool Full() {
1433  return (mSize == mCapacity);
1434  }
1435 
1436  int Size() {
1437  return mSize;
1438  }
1439 
1440  int Capacity() {
1441  return mCapacity;
1442  }
1443 
1444  int mFront; // front entry
1445  int mBack; // back entry
1446  int mSize; // count of entries in the ring
1447  int mCapacity; // ring capacity
1448  array<WaveformDataPoints^>^ mRing;
1449  System::Threading::Mutex^ mMutex;
1450  };
